1. install stuff 

	ifconfig eth0 up
	udhcpc eth0
	setup-apkrepos
	apk add e2fsprogs grub

1. make partition with fdisk
2. format

	mkfs.ext4 /dev/sda1

3. mount /dev/sda1 /mnt

4. run setup-alpine
5. export BOOTLOADER=grub
6. setup-disk /mnt

	https://wiki.alpinelinux.org/wiki/Setting_up_disks_manually
